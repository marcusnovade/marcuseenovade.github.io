  /**
  * 
  * @param {*} eTest 
  */
 function generateExpects (eTest) {
   var str = "//------- Excepts -------- \n";
   for (const spi of eTest) {
     var param = JSON.stringify(spi.spy.calls.allArgs(), (key, value) => typeof value === "undefined" ? null : value);
     var exceptStr = `expect(${spi.name}.calls.allArgs()).toEqual(
         ${param}
 );`;
     str += exceptStr + "\n";
   }
   str += "//------- End -------- \n";
 
   console.log(str);
 }
